\question Let $A=\left[ \begin{array}{rr} 1&2\\2&-1\\1&-1\\-1&-1  \end{array} \right] $

\begin{parts}
\part Find a basis for the Col($A$).
\part Find a basis for the Null($A$).
\part Find a basis for Row($A$).
\part Find a basis for the Col($A^{\text{T}}$).
\part Find a basis for the Null($A^{\text{T}}$).
\part Find a basis for Row($A^{\text{T}}$).
\end{parts}


\question Let $T$ be the transformation given by the matrix (having associated/standard matrix):

\[ A=\left[ \begin{array}{rr} 1&2\\2&-1\\1&-1\\-1&-1  \end{array} \right] \]

(same as in the previous problem).

\begin{parts}
\part What is the domain of $T$?
\part Find a basis for the domain of $T$. (Hint: You do not need to do any work, just pick a trivial basis!)
\part What is the codomain of $T$?
\part Find a basis for the codomain of $T$. (Hint: You do not need to do any work, just pick a trivial basis!)
\part Find a basis for the image of $T$.  (You do not need to do any work, just use the previous problem).
\part Find a basis for the kernel of $T$.  (You do not need to do any work, just use the previous problem).
\end{parts}

\question Let 

\[X = \left( \left[ \begin{array}{r} 1 \\ 1   \end{array} \right], \left[ \begin{array}{r} 0 \\  1 \end{array} \right] \right) \]

\begin{parts}
\part Show $X$ is an ordered basis of $\mathbb{R}^2$.
\part Let $K$ be the coordinate transformation defined by the ordered basis $X$.  The vector ${\bf w} = \left[ \begin{array}{r} -2 \\ 1   \end{array} \right]$ in in $\mathbb{R}^2$.  Calculate $K({\bf w})$.
\end{parts}

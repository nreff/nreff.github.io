\begin{choices}
\choice $x_1 + 2x_2 + x_4$
\choice $x_2 + x_3 + x_4$
\choice $x_1-x_2+x_3+2x_4$
\choice $2x_1 + 2\sqrt{x_2} + 3(x_3)^{(1/3)}+x_4$
\choice $x_1+x_2+x_3+\pi x_4$
\end{choices}
